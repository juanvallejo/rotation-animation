			var isRotating = false;
			$('.gear').on('mouseenter',function() {
			  if(!isRotating) {
			    isRotating = true;
			    $('#gr_one').rotate(360,{speed:250},function() {
			      $(this).rotate(0);
			      if(isRotating) isRotating = false;
			    });
			    $('#gr_two').rotate(360,{speed:350},function() {
			      $(this).rotate(0);
			      if(isRotating) isRotating = false;
			    });
			  }
			});

			    $('#gear_one').rotate(0,{speed:400});

			      if(isRotating) isRotating = false;
			    });
			  }
